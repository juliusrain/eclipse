// Protocol for game data transfer

// Syntax of message (JSON object)
    {
        'action': 'see below'
        'sender': 'name'
        'body': object
    }

// Syntax of 'body' (JSON object)

// CHAT
    {
        'time': 'H:M'
        'message': 'text'
    }

//ACTIONS:
    // these are to be sent to all players
    CHAT - chat message

    // these are sent to other players at the same planet
    POS - position update
    NEW - appearance of new object (player)

    // these are commands to interact with the database
    RETR - retrieve objects data
    SAVE - save object data (append and update data)
    OVER - overwrite data (whatever's passed becomes the data)
